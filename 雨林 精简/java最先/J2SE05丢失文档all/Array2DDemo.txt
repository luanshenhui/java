package com.yuin.am;
import java.util.*;
public class Array2DDemo {
	public static void main(String[] args){
		int[][]arr1=new int[4][5];
		arr1[2][3]=1;
		for(int i =0;i<arr1.length;i++){
			for(int j=0;j<arr1[0].length;j++){//arr1[0]第一个小数组
				System.out.print(arr1[i][j]);
			}
			System.out.println();
		}
		//System.out.println(Arr)
	}

}
