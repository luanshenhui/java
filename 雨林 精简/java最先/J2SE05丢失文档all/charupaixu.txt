package lianxi;

import java.util.Arrays;

public class charupaixu {
	public static void main(String[] args){
		int a[]={6,8,9,3,5};
		int i,j;
		int k;
		for(j=1;j<a.length;j++){
			k=a[j];
			i=j-1;//将手里的值琢一和前面比较，前一张派的的下标
			while(i>=0&&a[i]>k){
				a[i]=a[i+1];
				i--;
			}
			a[i+1]=k;//循环不成立，说明这是最大值，直接放在档期位置
		}
		System.out.print(Arrays.toString(a));
	}

}
