	public Emp(int empID, String name, int salary, int age, String hireDate) {
		// TODO Auto-generated constructor stub
		super();
		this.empID = empID;
		this.name = name;
		//this.deptId = deptId;

		this.age = age;
		this.hireDate = hireDate;
		
		
	}

	@Override
	public String toString() {
		return "Emp [Salary=" + Salary + ", age=" + age + ", deptId=" + deptId

				+ name + "]";
	}

	@Override
	public boolean equals(Object o) {
		/*
		 * 1不为null 2类型匹配 3自定义规则
		 */

		if (o == null) {
			return false;
		}
		if (o instanceof Emp) {
			Emp e = (Emp) o;
			if (this.empID == e.empID && this.name.equals(e.name)) {
				return true;
			}
		}
		return false;
	}

	@Override
	public int hashCode() {
		return this.empID * 1001;
	}
